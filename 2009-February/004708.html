This script makes "svn st" be silent:


#!/bin/sh

{
echo Makefile 
echo Makefile.in
echo openocd.aux
echo openocd.cp
echo openocd.fn
echo openocd.pg
echo openocd.pdf
echo openocd.log
echo openocd.tp
echo openocd.ky
echo openocd.cps
echo openocd.toc
echo openocd.vr
echo version.texi
echo texinfo.tex
echo mdate-sh
echo stamp-vti
echo openocd.info
} >.ignore
svn propset svn:ignore -F .ignore doc

{
echo Makefile
echo Makefile.in
echo .deps
} >.ignore
svn propset svn:ignore -F .ignore src/server src/svf src/flash 
src/xsvf
svn propset svn:ignore -F .ignore src/target src/jtag src/pld

{
echo Makefile
echo Makefile.in
echo .deps
echo startup_tcl.c
echo bin2char
} >.ignore
svn propset svn:ignore -F .ignore src/helper

{
echo Makefile
echo Makefile.in
echo .deps
echo openocd
echo openocd.cfg
} >.ignore
svn propset svn:ignore -F .ignore src

{
echo Makefile
echo Makefile.in
echo stamp-h1
echo config.status
echo config.guess
echo config.sub
echo aclocal.m4
echo install-sh
echo config.h
echo compile
echo config.log
echo depcomp
echo autom4te.cache
echo missing
echo configure
echo config.h.in
echo patches
echo .pc
} >.ignore
svn propset svn:ignore -F .ignore .

rm .ignore
