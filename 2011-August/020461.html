Replaced the following line and working out of tree - not tested inline

	if test -d $(srcdir)/.git -a \( ! -e $(distdir)/ChangeLog -o -w
$(distdir)/ChangeLog \) ; then \
		git --git-dir $(srcdir)/.git log | $(srcdir)/tools/git2cl/git2cl >
$(distdir)/ChangeLog ; \
	fi
