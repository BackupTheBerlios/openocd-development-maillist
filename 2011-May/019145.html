	{ NAND_MFR_SAMSUNG,	0xD5, 0, 8192, 0, LP_OPTIONS, 	"K9GAG08 2GB NAND 3.3V x8 (MLC 2bit/cell)"},
	{ NAND_MFR_SAMSUNG,	0xD7, 0, 8192, 0, LP_OPTIONS, 	"K9LBG08/K9HCG08 4/8GB NAND 3.3V x8 (MLC 2bit/cell)"},
	/* start "museum" IDs */
	{ 0x0,			0x6e, 256, 1, 0x1000, 0, 	"NAND 1MiB 5V 8-bit"},
	{ 0x0,			0x64, 256, 2, 0x1000, 0,	"NAND 2MiB 5V 8-bit"},

- description moved to the end for text alignment reasons
